```js
/**
 * @param {number} n
 * @return {number}
 */
// 简单dp
// 一次可以走一阶或者两阶，这样对某一阶而言，要么从一阶前走来，要么从二阶前走来，二者相加即为走来的总可能数
const climbStairs = (n) => {
    // 将dp数组优化为两个值，dp[i-1]和dp[i-2]
    let dp1, dp2;
    // const dp = new Array(n + 1).fill(0);
    // 到第0阶
    dp1 = 1;
    // 0 => 1
    dp2 = 1;
    for (let i = 2; i <= n; i++) {
        dp2 += dp1;
        [dp1, dp2] = [dp2, dp1];
        // dp[i] = dp[i - 1] + dp[i - 2];
    }
    // return dp[n];
    return dp1;
};
```
